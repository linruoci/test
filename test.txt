CREATE TABLE WEBHOOK_REQUEST_HISTORY (
    ID VARCHAR2(36) PRIMARY KEY,
    METHOD VARCHAR2(10),
    PATH VARCHAR2(500),
    FULL_URL VARCHAR2(1000),
    TIMESTAMP TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    IP VARCHAR2(50),
    HOST VARCHAR2(255),
    SIZE VARCHAR2(50),
    PROCESSING_TIME VARCHAR2(50),
    NOTE VARCHAR2(500),
    HEADERS CLOB,
    BODY CLOB,
    QUERY CLOB,
    WEBHOOK_URL_ID VARCHAR2(36),
    CONSTRAINT FK_WEBHOOK_URL FOREIGN KEY (WEBHOOK_URL_ID) REFERENCES WEBHOOK_URL(ID)
);
